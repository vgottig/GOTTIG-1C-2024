73void distancias_task(void *pvParameter)
74{
75 uint16_t distancia;
76
77 while (1)

80

82 {
83
84 distancia = HcSr04ReadDistanceInCentimeters();
85
86 if (!tecla2) // Si no apreto la tecla2
87 {
88 LcdItsE0803Write(distancia); // Pantalla LCD
89 }
90
91 UartSendString(UART_PC, (char *)UartItoa(distancia, 10)); // Lo que mido, lo convierto a string y lo muestro en la PC
92 UartSendString(UART_PC, " cm\n\r");
93
94 if (distancia < 10)
95 {
96 LedsOffAll();
97 }
98
99 else if (distancia > 10 && distancia < 20)
100 {
101 LedOn(LED_1);
102 LedOff(LED_2);
103 LedOff(LED_3);
104 }
105
106 else if (distancia > 20 && distancia < 30)
107 {
108 LedOn(LED_1);
109 LedOn(LED_2);
110 LedOff(LED_3);
111 }
112
113 else if (distancia > 30)
114 {
115 LedOn(LED_1);
116 LedOn(LED_2);
117 LedOn(LED_3);
118 }
119 }
120 else
121 {
122 LcdItsE0803Off();
123 LedsOffAll();
124 }
125 }
126}
